===== /root/.hermes/scripts/gbrain_maintenance_self_os.sh =====
#!/usr/bin/env bash
set -euo pipefail

LOG_FILE="/var/log/gbrain-maintenance.log"
GBRAIN_BIN="/root/.bun/bin/gbrain"
SELF_OS_DIR="/data/Self-OS"

export HOME="/root"
export PATH="/root/.bun/bin:${PATH:-}"

log() {
  printf '[%s] %s\n' "$(date -Is)" "$*" >> "${LOG_FILE}"
}

run_logged() {
  log "+ $*"
  "$@" >> "${LOG_FILE}" 2>&1
}

run_embedding_refresh() {
  local tmp
  tmp="$(mktemp)"
  log "+ ${GBRAIN_BIN} embed --stale"
  if "${GBRAIN_BIN}" embed --stale >"${tmp}" 2>&1; then
    cat "${tmp}" >> "${LOG_FILE}"
    rm -f "${tmp}"
    return 0
  fi

  cat "${tmp}" >> "${LOG_FILE}"
  if grep -q "OpenAI embedding requires OPENAI_API_KEY" "${tmp}"; then
    log "soft-block: stale embeddings skipped because OPENAI_API_KEY is not configured for GBrain embeddings"
    rm -f "${tmp}"
    return 0
  fi

  rm -f "${tmp}"
  return 1
}

{
  log "Starting GBrain maintenance for ${SELF_OS_DIR}"

  if [[ ! -x "${GBRAIN_BIN}" ]]; then
    log "fatal: ${GBRAIN_BIN} not found or not executable"
    exit 1
  fi
  if [[ ! -d "${SELF_OS_DIR}/.git" ]]; then
    log "fatal: ${SELF_OS_DIR} is not a git repo"
    exit 1
  fi

  cd "${SELF_OS_DIR}"

  run_logged "${GBRAIN_BIN}" apply-migrations --yes
  run_logged "${GBRAIN_BIN}" skillpack check
  run_embedding_refresh
  run_logged "${GBRAIN_BIN}" extract links --dir "${SELF_OS_DIR}"
  run_logged "${GBRAIN_BIN}" extract timeline --dir "${SELF_OS_DIR}"
  run_logged "${GBRAIN_BIN}" features --json

  log "Completed GBrain maintenance for ${SELF_OS_DIR}"
} >> "${LOG_FILE}" 2>&1
